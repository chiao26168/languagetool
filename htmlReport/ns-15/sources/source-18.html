   @Override
   public String toAdvancedTypography(String input) {
     String output = super.toAdvancedTypography(input);
     //non-breaking space
     output = output.replaceAll("\\b([a-zA-Z]\\.)([a-zA-Z]\\.)", "$1\u00a0$2");
     output = output.replaceAll("\\b([a-zA-Z]\\.)([a-zA-Z]\\.)", "$1\u00a0$2");
     return output;
   }
